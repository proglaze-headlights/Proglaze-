    <script type="module">
        import { initializeApp } from "https://www.gstatic.com/firebasejs/11.6.1/firebase-app.js";
        import { getAuth, signInWithCustomToken, signInAnonymously } from "https://www.gstatic.com/firebasejs/11.6.1/firebase-auth.js";
        import { getFirestore, collection, addDoc, serverTimestamp } from "https://www.gstatic.com/firebasejs/11.6.1/firebase-firestore.js";
        import { setLogLevel } from "https://www.gstatic.com/firebasejs/11.6.1/firebase-firestore.js";
        
        // Set Firestore log level to Debug
        setLogLevel('Debug');

        // Configuration Constants
        const EMAIL_RECEIVER = "chazwill31@outlook.com";
        const EMAIL_ENDPOINT = `https://formsubmit.co/ajax/${EMAIL_RECEIVER}`; // Using AJAX for non-redirect submission
        const MAX_RETRIES = 5;
        const BASE_DELAY = 1000; // 1 second

        // Global environment variables (MANDATORY USE)
        const appId = typeof __app_id !== 'undefined' ? __app_id : 'proglaze-default-id';
        const firebaseConfig = typeof __firebase_config !== 'undefined' ? JSON.parse(__firebase_config) : null;
        const initialAuthToken = typeof __initial_auth_token !== 'undefined' ? __initial_auth_token : null;

        let app, db, auth;
        let userId = null;
        
        // Booking State - Updated to include image details
        let bookingState = {
            fileCount: 0,
            fileNames: [],
            tierName: null,
            tierPrice: 0,
            locationZone: null,
            travelFee: 0,
            total: 0
        };

        // DOM elements for state management
        const bookButton = document.getElementById('book-button');
        const continueButton = document.getElementById('continueButton');
        const uploadText = document.getElementById('uploadText');
        const fileList = document.getElementById('fileList');
        const imageValidationMessage = document.getElementById('imageValidationMessage');
        const buttonText = document.getElementById('button-text');
        const loadingSpinner = document.getElementById('loading-spinner');
        const fleetButton = document.getElementById('fleet-button');
        const fleetButtonText = document.getElementById('fleet-button-text');
        const fleetLoadingSpinner = document.getElementById('fleet-loading-spinner');


        // Utility to show custom modal
        function showModal(title, message, color) {
            const modal = document.getElementById('feedbackModal');
            document.getElementById('modalTitle').textContent = title;
            document.getElementById('modalTitle').classList.remove('text-pro-red', 'text-green-500');
            document.getElementById('modalTitle').classList.add(color === 'green' ? 'text-green-500' : 'text-pro-red');
            document.getElementById('modalMessage').textContent = message;
            modal.classList.remove('hidden');
            modal.classList.add('flex');
        }

        // Handle button loading state (B2C)
        function setButtonLoading(isLoading) {
            if (isLoading) {
                bookButton.disabled = true;
                bookButton.classList.add('opacity-50', 'cursor-not-allowed');
                buttonText.textContent = "Processing...";
                loadingSpinner.classList.remove('hidden');
            } else {
                // Only re-enable if valid state is met (checked in updateSummary)
                updateSummary(); 
                buttonText.textContent = "Submit Booking Inquiry";
                loadingSpinner.classList.add('hidden');
            }
        }
        
        // Handle button loading state (B2B)
        function setFleetButtonLoading(isLoading) {
            if (isLoading) {
                fleetButton.disabled = true;
                fleetButton.classList.add('opacity-50', 'cursor-not-allowed');
                fleetButtonText.textContent = "Processing...";
                fleetLoadingSpinner.classList.remove('hidden');
            } else {
                fleetButton.disabled = false;
                fleetButton.classList.remove('opacity-50', 'cursor-not-allowed');
                fleetButtonText.textContent = "Submit Fleet Inquiry";
                fleetLoadingSpinner.classList.add('hidden');
            }
        }

        // Firebase Initialization and Authentication
        async function initializeFirebase() {
            try {
                if (firebaseConfig) {
                    app = initializeApp(firebaseConfig);
                    db = getFirestore(app);
                    auth = getAuth(app);
                    
                    // Auth using custom token if available, otherwise anonymously
                    if (initialAuthToken) {
                        await signInWithCustomToken(auth, initialAuthToken);
                    } else {
                        await signInAnonymously(auth);
                    }
                    userId = auth.currentUser?.uid || crypto.randomUUID();
                } else {
                    console.error("Firebase config is missing.");
                }
            } catch (error) {
                console.error("Firebase initialization failed: ", error);
            }
        }
        
        // Function to save data to Firestore (Generic)
        // collectionName is either 'proglaze_bookings' (B2C) or 'proglaze_fleet_leads' (B2B)
        async function saveData(data, collectionName) {
            if (!db || !userId) {
                throw new Error("Firestore not initialized or user not authenticated. Cannot save data.");
            }
            
            // Collection path: /artifacts/{appId}/public/data/{collectionName}
            const collectionPath = `artifacts/${appId}/public/data/${collectionName}`;
            
            for (let i = 0; i < MAX_RETRIES; i++) {
                try {
                    await addDoc(collection(db, collectionPath), {
                        ...data,
                        userId: userId,
                        timestamp: serverTimestamp()
                    });
                    return true; // Success
                } catch (e) {
                    console.error(`Firestore attempt ${i + 1} failed: `, e);
                    if (i < MAX_RETRIES - 1) {
                        await new Promise(resolve => setTimeout(resolve, BASE_DELAY * Math.pow(2, i)));
                    } else {
                        throw new Error(`Failed to submit data to ${collectionName} after multiple retries.`);
                    }
                }
            }
        }
        
        // Function to send email notification using FormSubmit endpoint (Generic)
        async function sendNotification(data, type) {
            const formData = new FormData();
            const isB2C = type === 'B2C';
            let subject = "";

            // Add all data fields
            for (const key in data) {
                if (data[key] !== null) {
                    formData.append(key, data[key]);
                }
            }
            
            if (isB2C) {
                subject = `PROGLAZE New Booking Inquiry - R${data.total} (${data.locationZone})`;
                const fileListFormatted = data.imageFilenames.length > 0 ? data.imageFilenames.join(', ') : 'No files attached (Client confirmed upload intention)';
                formData.append("Image Filenames", fileListFormatted);
                formData.append("--- NOTE ---", "The physical images were not transmitted via this email. Client confirms files were ready at the time of booking. Contact client for image transmission.");
            } else { // B2B
                subject = `PROGLAZE NEW FLEET LEAD: ${data.companyName} (${data.fleetSize})`;
            }

            formData.append("_subject", subject);
            
            for (let i = 0; i < MAX_RETRIES; i++) {
                try {
                    const response = await fetch(EMAIL_ENDPOINT, {
                        method: 'POST',
                        body: formData,
                        headers: { 'Accept': 'application/json' }
                    });

                    if (response.ok) return true;
                } catch (e) {
                    console.error(`Email attempt ${i + 1} fetch failed for type ${type}: `, e);
                }

                if (i < MAX_RETRIES - 1) {
                    await new Promise(resolve => setTimeout(resolve, BASE_DELAY * Math.pow(2, i)));
                } else {
                    throw new Error(`Failed to send ${type} notification after multiple retries.`);
                }
            }
        }
        
        // Handle Image Upload (Step 1 - B2C)
        window.handleImageUpload = (event) => {
            const files = event.target.files;
            
            if (files.length === 0) {
                bookingState.fileCount = 0;
                bookingState.fileNames = [];
                uploadText.textContent = "Upload 2-3 Photos Now";
                fileList.innerHTML = '';
                imageValidationMessage.classList.remove('hidden');
            } else {
                bookingState.fileCount = files.length;
                bookingState.fileNames = Array.from(files).map(f => f.name);
                
                uploadText.textContent = `${files.length} File(s) Selected`;
                fileList.innerHTML = bookingState.fileNames.map(name => 
                    `<p class="text-green-600 flex items-center"><svg class="h-4 w-4 mr-2" fill="currentColor" viewBox="0 0 20 20"><path d="M9 2a2 2 0 00-2 2v2H5a2 2 0 00-2 2v10a2 2 0 002 2h10a2 2 0 002-2V8a2 2 0 00-2-2h-2V4a2 2 0 00-2-2H9zm2 6h3v8H7V8h3zm-3-4a1 1 0 011-1h2a1 1 0 011 1v2h-4V4z" clip-rule="evenodd" fill-rule="evenodd"></path></svg>${name} (${(files[Array.from(files).findIndex(f => f.name === name)].size / 1024 / 1024).toFixed(2)} MB)</p>`
                ).join('');
                imageValidationMessage.classList.add('hidden');
            }

            updateSummary();
            updateContinueButton();
        };

        function updateContinueButton() {
             if (bookingState.fileCount > 0) {
                continueButton.classList.remove('bg-gray-300', 'text-gray-700', 'pointer-events-none', 'opacity-50');
                continueButton.classList.add('btn-red', 'text-pro-white', 'shadow-lg');
                continueButton.disabled = false;
            } else {
                continueButton.classList.remove('btn-red', 'text-pro-white', 'shadow-lg');
                continueButton.classList.add('bg-gray-300', 'text-gray-700', 'pointer-events-none', 'opacity-50');
                continueButton.disabled = true;
            }
        }


        // Handle Tier Selection (Step 2 - B2C)
        window.selectTier = (name, price, event) => {
            bookingState.tierName = name;
            bookingState.tierPrice = price;

            // Highlight the selected button and remove highlight from others
            const buttons = document.querySelectorAll('#tiers button');
            buttons.forEach(btn => btn.closest('div').classList.remove('ring-4', 'ring-pro-red', 'font-bold'));
            
            // Find the clicked button's container and highlight it
            event.currentTarget.closest('div').classList.add('ring-4', 'ring-pro-red', 'font-bold');

            updateSummary();
        };

        // Handle Location Selection (Step 3 - B2C)
        window.selectLocation = (zone, fee, event) => {
            bookingState.locationZone = zone;
            bookingState.travelFee = fee;

            // Highlight the selected zone button
            const zoneButtons = document.querySelectorAll('#zone-container button');
            zoneButtons.forEach(btn => btn.classList.remove('selected'));
            
            if (event.target.tagName === 'BUTTON') {
                event.target.classList.add('selected');
            } else {
                 event.target.closest('button').classList.add('selected');
            }
            
            document.getElementById('zone-fee-info').textContent = `Travel Fee: R${fee} (${zone})`;
            document.getElementById('zone-feedback').classList.add('hidden');

            updateSummary();
        };

        // Update Summary and Button State (B2C)
        function updateSummary() {
            bookingState.total = bookingState.tierPrice + bookingState.travelFee;

            // Update display fields
            document.getElementById('summary-images').textContent = `${bookingState.fileCount} file(s) ready`;
            document.getElementById('summary-tier').textContent = bookingState.tierName || 'N/A';
            document.getElementById('summary-price').textContent = `R${bookingState.tierPrice}`;
            document.getElementById('summary-location').textContent = bookingState.locationZone || 'N/A';
            document.getElementById('summary-fee').textContent = `R${bookingState.travelFee}`;
            document.getElementById('summary-total').textContent = `R${bookingState.total}`;
            
            const validationMessage = document.getElementById('validation-message');

            // Enable/Disable button based on all selections (Image, Tier, Zone)
            if (bookingState.fileCount > 0 && bookingState.tierName && bookingState.locationZone) {
                bookButton.disabled = false;
                bookButton.classList.remove('opacity-50', 'cursor-not-allowed');
                validationMessage.classList.add('hidden');
            } else {
                bookButton.disabled = true;
                bookButton.classList.add('opacity-50', 'cursor-not-allowed');
                validationMessage.classList.remove('hidden');
            }
        }
        
        // Handle B2C Form Submission (DB Save + Email Notification)
        window.handleBookingSubmit = async (event) => {
            event.preventDefault();
            setButtonLoading(true);

            if (bookingState.fileCount === 0 || !bookingState.tierName || !bookingState.locationZone) {
                document.getElementById('validation-message').classList.remove('hidden');
                setButtonLoading(false);
                return;
            }

            const fullName = document.getElementById('fullName').value.trim();
            const contactNumber = document.getElementById('contactNumber').value.trim();
            const address = document.getElementById('address').value.trim();
            
            if (!fullName || !contactNumber || !address) {
                showModal("Validation Error", "Please fill in all contact details (Name, Contact, Address).", 'pro-red');
                setButtonLoading(false);
                return;
            }

            // Prepare final data object for Firestore and Email
            const bookingData = {
                fullName: fullName,
                contactNumber: contactNumber,
                serviceAddress: address,
                imageFilesCount: bookingState.fileCount,
                imageFilenames: bookingState.fileNames,
                ...bookingState, // tierName, tierPrice, locationZone, travelFee, total
            };

            try {
                // 1. Save to Firestore (B2C: proglaze_bookings)
                await saveData(bookingData, 'proglaze_bookings');

                // 2. Send Email Notification (B2C)
                await sendNotification(bookingData, 'B2C');
                
                // 3. Success Feedback and Reset
                // Clear form and state after success
                document.getElementById('fullName').value = '';
                document.getElementById('contactNumber').value = '';
                document.getElementById('address').value = '';
                document.getElementById('imageUpload').value = ''; 
                
                // Reset state display (visual and internal)
                bookingState = { fileCount: 0, fileNames: [], tierName: null, tierPrice: 0, locationZone: null, travelFee: 0, total: 0 };
                updateSummary();
                updateContinueButton();
                uploadText.textContent = "Upload 2-3 Photos Now";
                fileList.innerHTML = '';
                
                showModal("Booking Submitted!", 
                          `Thank you, ${bookingData.fullName}. Your ${bookingData.tierName} inquiry for R${bookingData.total} is logged and your images are ready for assessment. A PROGLAZE technician will contact you shortly.`, 
                          'green');
                          
            } catch (e) {
                console.error("B2C Booking process failed: ", e);
                showModal("System Error", "A critical error occurred during B2C submission. Please check the console for details and try again.", 'pro-red');
            } finally {
                setButtonLoading(false);
            }
        };

        // Handle B2B Form Submission (DB Save + Email Notification)
        window.handleFleetSubmit = async (event) => {
            event.preventDefault();
            setFleetButtonLoading(true);

            const companyName = document.getElementById('companyName').value.trim();
            const contactPerson = document.getElementById('contactPerson').value.trim();
            const contactNumber = document.getElementById('fleetContactNumber').value.trim();

                fleetSize,

                await saveData(fleetData, 'proglaze_fleet_leads');

                // 3. Success Feedback and Reset

                          'green');
                
            } catch (e) {
                console.error("B2B Fleet submission failed: ", e);
                showModal("System Error", "A critical error occurred during B2B submission. Please check the console for details and try again.", 'pro-red');
            } finally {
                setFleetButtonLoading(false);
            }
        };
        
        // Initialize everything on load
        window.onload = () => {
             initializeFirebase();
             updateContinueButton(); // Initial state check
             updateSummary(); // Initial state check
        }
